-- 1. Retrieve customers who ordered all available products. 

SELECT c.customer_id, c.name
FROM customers c
JOIN orders o ON c.customer_id = o.customer_id
JOIN order_items oi ON o.order_id = oi.order_id
GROUP BY c.customer_id, c.name
HAVING COUNT(DISTINCT oi.product_id) = (SELECT COUNT(*) FROM products);


-- 2. List products that have never appeared in any order.

SELECT p.product_id, p.product_name
FROM products p
WHERE NOT EXISTS (
    SELECT 1
    FROM order_items oi
    WHERE oi.product_id = p.product_id
);


-- 3. Find orders that contain only one item, but that item is the most expensive product.

SELECT order_id
FROM order_items
WHERE product_id = (
    SELECT product_id
    FROM products
    WHERE price = (SELECT MAX(price) FROM products)
)
GROUP BY order_id
HAVING COUNT(*) = 1;


-- 4. Identify customers whose first-ever order has more than 3 items.

SELECT c.customer_id, c.name
FROM (
    SELECT o.customer_id,
           o.order_id,
           ROW_NUMBER() OVER (PARTITION BY o.customer_id ORDER BY o.order_date) AS rn
    FROM orders o
) first_order
JOIN order_items oi ON first_order.order_id = oi.order_id
JOIN customers c ON first_order.customer_id = c.customer_id
WHERE rn = 1
GROUP BY c.customer_id, c.name
HAVING COUNT(*) > 3;


-- 5. Find customers who placed orders but never made a payment.

SELECT DISTINCT c.customer_id, c.name
FROM customers c
JOIN orders o ON c.customer_id = o.customer_id
WHERE NOT EXISTS (
    SELECT 1
    FROM payments p
    WHERE p.order_id = o.order_id
);


-- 6. Retrieve orders that include both Mouse and Keyboard.

SELECT o.order_id
FROM orders o
WHERE EXISTS (
    SELECT 1
    FROM order_items oi
    JOIN products p ON oi.product_id = p.product_id
    WHERE oi.order_id = o.order_id
    AND LOWER(p.product_name) LIKE '%mouse%'
)
AND EXISTS (
    SELECT 1
    FROM order_items oi
    JOIN products p ON oi.product_id = p.product_id
    WHERE oi.order_id = o.order_id
    AND LOWER(p.product_name) LIKE '%keyboard%'
);


-- 7. For each customer, list the total amount ordered compared with their total payments.

WITH order_totals AS (
    SELECT o.customer_id,
           SUM(oi.quantity * p.price) AS total_ordered
    FROM orders o
    JOIN order_items oi ON o.order_id = oi.order_id
    JOIN products p ON oi.product_id = p.product_id
    GROUP BY o.customer_id
),
payment_totals AS (
    SELECT o.customer_id,
           SUM(p.amount) AS total_paid
    FROM orders o
    JOIN payments p ON o.order_id = p.order_id
    GROUP BY o.customer_id
)
SELECT c.customer_id,
       c.name,
       NVL(ot.total_ordered,0) AS total_ordered,
       NVL(pt.total_paid,0) AS total_paid
FROM customers c
LEFT JOIN order_totals ot ON c.customer_id = ot.customer_id
LEFT JOIN payment_totals pt ON c.customer_id = pt.customer_id;


-- 8. List orders where the payment amount is less than the order value.

SELECT o.order_id,
       SUM(oi.quantity * p.price) AS total_order_value,
       NVL(SUM(pay.amount),0) AS total_paid
FROM orders o
JOIN order_items oi ON o.order_id = oi.order_id
JOIN products p ON oi.product_id = p.product_id
LEFT JOIN payments pay ON o.order_id = pay.order_id
GROUP BY o.order_id
HAVING NVL(SUM(pay.amount),0) < SUM(oi.quantity * p.price);


-- 9. Identify products that appear in orders but have zero inventory.

SELECT DISTINCT p.product_id, p.product_name
FROM products p
JOIN order_items oi ON p.product_id = oi.product_id
JOIN inventory i ON p.product_id = i.product_id
WHERE i.quantity_on_hand = 0;


-- 10. Find all orders where every product in the order has quantity_on_hand < 20.

SELECT oi.order_id
FROM order_items oi
JOIN inventory i ON oi.product_id = i.product_id
GROUP BY oi.order_id
HAVING MAX(i.quantity_on_hand) < 20;

-- 11. List customers who ordered a product before its inventory was last updated.

SELECT C.CUSTOMER_ID, C.NAME, O.ORDER_ID, O.ORDER_DATE, I.PRODUCT_ID, I.LAST_UPDATED
FROM CUSTOMERS C
JOIN ORDERS O ON C.CUSTOMER_ID = O.CUSTOMER_ID
JOIN ORDER_ITEMS OI ON O.ORDER_ID = OI.ORDER_ID
JOIN INVENTORY I ON OI.PRODUCT_ID = I.PRODUCT_ID
WHERE O.ORDER_DATE < I.LAST_UPDATED;


-- 12. Retrieve orders containing products from at least 3 different categories.

SELECT o.order_id
FROM orders o
JOIN order_items oi ON o.order_id = oi.order_id
JOIN products p ON oi.product_id = p.product_id
GROUP BY o.order_id
HAVING COUNT(DISTINCT
    CASE
        WHEN LOWER(p.product_name) LIKE '%laptop%' THEN 'LAPTOP'
        WHEN LOWER(p.product_name) LIKE '%mouse%' THEN 'MOUSE'
        WHEN LOWER(p.product_name) LIKE '%keyboard%' THEN 'KEYBOARD'
        WHEN LOWER(p.product_name) LIKE '%monitor%' THEN 'MONITOR'
        ELSE 'OTHER'
    END
) >= 3;


-- 13. Find customers who placed orders in each month of a given year.

SELECT C.CUSTOMER_ID, C.NAME
FROM CUSTOMERS C
JOIN ORDERS O ON C.CUSTOMER_ID = O.CUSTOMER_ID
WHERE EXTRACT(YEAR FROM O.ORDER_DATE) = 2025
GROUP BY C.CUSTOMER_ID, C.NAME
HAVING COUNT(DISTINCT EXTRACT(MONTH FROM O.ORDER_DATE)) = 12;


-- 14. List orders where the highest-priced item is the only item ordered.

SELECT OI.ORDER_ID
FROM ORDER_ITEMS OI
JOIN PRODUCTS P ON OI.PRODUCT_ID = P.PRODUCT_ID
GROUP BY OI.ORDER_ID
HAVING COUNT(*) = 1
AND MAX(P.PRICE) = (SELECT MAX(PRICE) FROM PRODUCTS);


-- 15. Retrieve top 3 customers with the most distinct products ordered.

SELECT CUSTOMER_ID, NAME, DISTINCT_PRODUCTS
FROM (
    SELECT C.CUSTOMER_ID, C.NAME,
           COUNT(DISTINCT OI.PRODUCT_ID) AS DISTINCT_PRODUCTS,
           DENSE_RANK() OVER (ORDER BY COUNT(DISTINCT OI.PRODUCT_ID) DESC) AS RNK
    FROM CUSTOMERS C
    JOIN ORDERS O ON C.CUSTOMER_ID = O.CUSTOMER_ID
    JOIN ORDER_ITEMS OI ON O.ORDER_ID = OI.ORDER_ID
    GROUP BY C.CUSTOMER_ID, C.NAME
)
WHERE RNK <= 3;


-- 16. Identify payments made after 48 hours from order_date.

SELECT P.PAYMENT_ID, P.ORDER_ID, O.ORDER_DATE, P.PAYMENT_DATE
FROM PAYMENTS P
JOIN ORDERS O ON P.ORDER_ID = O.ORDER_ID
WHERE (P.PAYMENT_DATE - O.ORDER_DATE) > 2;


-- 17. Find duplicate product purchases within the same order.

SELECT ORDER_ID, PRODUCT_ID, COUNT(*) AS DUPLICATE_COUNT
FROM ORDER_ITEMS

-- 18. List products ordered consistently in quantities > 1.

SELECT P.PRODUCT_ID, P.PRODUCT_NAME
FROM PRODUCTS P
JOIN ORDER_ITEMS OI ON P.PRODUCT_ID = OI.PRODUCT_ID

-- 19. Find inventory not used in orders for over 90 days.

SELECT P.PRODUCT_ID, P.PRODUCT_NAME, I.QUANTITY_ON_HAND, I.LAST_UPDATED
FROM PRODUCTS P
JOIN INVENTORY I ON P.PRODUCT_ID = I.PRODUCT_ID
WHERE P.PRODUCT_ID NOT IN (
    SELECT OI.PRODUCT_ID
    FROM ORDER_ITEMS OI
    JOIN ORDERS O ON OI.ORDER_ID = O.ORDER_ID
    WHERE O.ORDER_DATE >= SYSDATE - 90
);


-- 20. Identify orders where payment_method is NULL.

21. Using a CTE, compute total stock value for every product.

WITH stock_value AS (
    SELECT p.product_id,
           p.product_name,
           p.price,
           i.quantity_on_hand,
           (p.price * i.quantity_on_hand) AS total_value
    FROM products p
    JOIN inventory i
    ON p.product_id = i.product_id
)
SELECT * 
FROM stock_value;


22. Build a CTE to calculate monthly sales and find above-average months.

WITH monthly_sales AS (
    SELECT TO_CHAR(o.order_date,'YYYY-MM') AS month,
           SUM(p.price * oi.quantity) AS total_sales
    FROM orders o
    JOIN order_items oi ON o.order_id = oi.order_id
    JOIN products p ON oi.product_id = p.product_id
    GROUP BY TO_CHAR(o.order_date,'YYYY-MM')
)
SELECT *
FROM monthly_sales
WHERE total_sales > (SELECT AVG(total_sales) FROM monthly_sales);


23. Generate calendar dates using recursive CTE and join with orders.

WITH calendar_dates AS (
SELECT DATE '2025-01-01'+LEVEL-1 dt
FROM dual
CONNECT BY LEVEL<=365
)
SELECT c.dt,o.order_id
FROM calendar_dates c
LEFT JOIN orders o
ON c.dt=o.order_date;

26. Identify top 1% customers by spending using CTE.

WITH spending AS (
    SELECT c.customer_id,
           c.name,
           SUM(p.price * oi.quantity) AS total_spent
    FROM customers c
    JOIN orders o ON c.customer_id = o.customer_id
    JOIN order_items oi ON o.order_id = oi.order_id
    JOIN products p ON oi.product_id = p.product_id
    GROUP BY c.customer_id, c.name
)
SELECT *
FROM (
    SELECT s.*,
           PERCENT_RANK() OVER (ORDER BY total_spent DESC) AS pr
    FROM spending s
)
WHERE pr <= 0.01;


27. Find products with declining monthly order quantities.

WITH monthly_orders AS (
    SELECT product_id,
           TO_CHAR(o.order_date,'YYYY-MM') AS month,
           SUM(quantity) qty,
           LAG(SUM(quantity)) OVER (
               PARTITION BY product_id
               ORDER BY TO_CHAR(o.order_date,'YYYY-MM')
           ) AS prev_qty
    FROM order_items oi
    JOIN orders o ON oi.order_id = o.order_id
    GROUP BY product_id, TO_CHAR(o.order_date,'YYYY-MM')
)
SELECT *
FROM monthly_orders
WHERE qty < prev_qty;

28. Find inventory gaps using CTE to compare ordered vs on-hand quantities.

29. Compute running total of payments per customer using CTE.

-- 35. Find orders that exceed twice the customer’s average order size.

WITH order_totals AS (
    SELECT o.order_id,
           o.customer_id,
           SUM(oi.quantity * p.price) AS order_value
    FROM orders o
    JOIN order_items oi ON o.order_id = oi.order_id
    JOIN products p ON oi.product_id = p.product_id
    GROUP BY o.order_id, o.customer_id
),
customer_avg AS (
    SELECT customer_id,
           AVG(order_value) AS avg_order_value
    FROM order_totals
    GROUP BY customer_id
)
SELECT ot.order_id, ot.customer_id, ot.order_value
FROM order_totals ot
JOIN customer_avg ca ON ot.customer_id = ca.customer_id
WHERE ot.order_value > 2 * ca.avg_order_value;

-- 41. Create weekly sales buckets.

SELECT 
    TRUNC(o.order_date, 'IW') AS week_start,
    COUNT(DISTINCT o.order_id) AS orders_in_week,
    SUM(p.price * oi.quantity) AS total_sales,
    CASE 
        WHEN SUM(p.price * oi.quantity) > 50000 THEN 'High'
        WHEN SUM(p.price * oi.quantity) > 20000 THEN 'Medium'
        WHEN SUM(p.price * oi.quantity) > 5000 THEN 'Low'
        ELSE 'Very Low'
    END AS sales_bucket
FROM orders o
JOIN order_items oi ON o.order_id = oi.order_id
JOIN products p ON oi.product_id = p.product_id
GROUP BY TRUNC(o.order_date, 'IW')
ORDER BY week_start;

-- 44. Bucket customers by average order gap.

WITH order_gaps AS (
    SELECT o.customer_id,
           o.order_date,
           o.order_date - LAG(o.order_date) OVER (
               PARTITION BY o.customer_id ORDER BY o.order_date
           ) AS gap_days
    FROM orders o
),
avg_gaps AS (
    SELECT customer_id,
           AVG(gap_days) AS avg_gap
    FROM order_gaps
    GROUP BY customer_id
)
SELECT customer_id,
       avg_gap,
       CASE 
           WHEN avg_gap <= 30 THEN 'Frequent'
           WHEN avg_gap <= 90 THEN 'Regular'
           ELSE 'Rare'
       END AS gap_bucket
FROM avg_gaps;


-- 45. Rank customers by total ordered amount; return top 5 per email domain.

WITH customer_totals AS (
    SELECT c.customer_id,
           c.name,
           c.email,
           SUM(p.price * oi.quantity) AS total_spent,
           SUBSTR(c.email, INSTR(c.email, '@')) AS domain
    FROM customers c
    JOIN orders o ON c.customer_id = o.customer_id
    JOIN order_items oi ON o.order_id = oi.order_id
    JOIN products p ON oi.product_id = p.product_id
    GROUP BY c.customer_id, c.name, c.email
)
SELECT name,
       domain,
       total_spent
FROM (
    SELECT *,
           RANK() OVER (PARTITION BY domain ORDER BY total_spent DESC) AS rnk
    FROM customer_totals
)
WHERE rnk <= 5;

-- 51. Detect customers who stopped purchasing for 60 days via LEAD.

SELECT customer_id,
       order_date,
       next_order_date,
       gap_days
FROM (
    SELECT customer_id,
           order_date,
           LEAD(order_date) OVER (PARTITION BY customer_id ORDER BY order_date) AS next_order_date,
           LEAD(order_date) OVER (PARTITION BY customer_id ORDER BY order_date) - order_date AS gap_days
    FROM orders
)
WHERE gap_days > 60;


-- 52. Calculate percentage contribution of each item to order total.

SELECT oi.order_id,
       oi.product_id,
       (oi.quantity * p.price) AS item_value,
       SUM(oi.quantity * p.price) OVER (PARTITION BY oi.order_id) AS order_total,
       ROUND(
           (oi.quantity * p.price) /
           SUM(oi.quantity * p.price) OVER (PARTITION BY oi.order_id) * 100, 2
       ) AS percentage_contribution
FROM order_items oi
JOIN products p ON oi.product_id = p.product_id;

           PARTITION BY product_id
           ORDER BY total_quantity DESC
       ) AS rank_position
FROM (
    SELECT o.customer_id,
           oi.product_id,
           SUM(oi.quantity) AS total_quantity
    FROM orders o
    JOIN order_items oi ON o.order_id = oi.order_id
    GROUP BY o.customer_id, oi.product_id
);


-- 57. Find orders above the 90th percentile in value.

WITH order_values AS (
    SELECT oi.order_id,
           SUM(oi.quantity * p.price) AS order_value
    FROM order_items oi
    JOIN products p ON oi.product_id = p.product_id
    GROUP BY oi.order_id
)
SELECT order_id, order_value
FROM order_values
WHERE order_value >
(
    SELECT PERCENTILE_CONT(0.9) WITHIN GROUP (ORDER BY order_value)
    FROM order_values
);


-- 58. Split customers into quartiles based on spending.

WITH customer_spending AS (
    SELECT o.customer_id,
           SUM(oi.quantity * p.price) AS total_spending
    FROM orders o
    JOIN order_items oi ON o.order_id = oi.order_id
    JOIN products p ON oi.product_id = p.product_id
    GROUP BY o.customer_id
)
SELECT customer_id,
       total_spending,
       NTILE(4) OVER (ORDER BY total_spending DESC) AS spending_quartile
FROM customer_spending;
